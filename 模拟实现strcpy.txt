void my_strncpy(char* str, const char* ptr, int n)
{
	assert(str && ptr);
	int len = strlen(ptr);
	int p = n - len-1;
	if (len > n)
	{
		while (n)
		{
			*str++ = *ptr++;
			n--;
		}
		*str = '\0';
	}
	else if (len <= n)
	{
		while (*str++=*ptr++)
		{
			;
		}
		
		while (p)
		{
			*str = '\0';
			p--;
		}
	}
}
