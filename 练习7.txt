// 题目：企业发放的奖金更具利润提成。利润（i）低于等于10万元时，奖金可提10%；
//利润高于10万元低于20万元时，低于10万的按10% 提成，高于10万的按7.5 % 提成；
//20万到40万时, 高于20万的按5% 提成；
//40万到60万时，高于40万的按3 % 提成;
//60万到100万时，高于60万的按1.5 % 提成；
//高于100万时，超过100万的部分按1 % 提成。
//从键盘输入当月利润i，求应发放的奖金总数？

#define _CRT_SECURE_NO_WARNINGS 1
#include<stdio.h>
#include<string.h>
int  main()
{
    double i = 0;
    double ret = 0;
    printf("请输入当月的利润：");
    scanf("%lf", &i);
    if (i <= 100000)
    {
        ret =0.1 * i;
    }
    else if (100000 < i <= 200000)
    {
        ret = 0.075 * (i - 100000) + 0.1 * 100000;
    }
    else if (200000 < i <= 400000)
    {
        ret = 0.05 * (i - 200000) + 0.075 * 100000 + 0.1 *100000;
    }
    else if (400000 < i <= 600000)
    {
        ret = 0.03* (i - 400000) + 0.05* 200000 + 0.075* 100000 + 0.1* 100000;
    }
    else if (600000 < i <= 1000000)
    {
        ret = 0.015 * (i - 600000) +0.03 * 200000 +0.05* 200000 + 0.075* 100000 + 0.1 * 100000;
    }
    else
    {
        ret=0.01*(i-1000000)+ 0.015 * 400000 + 0.03* 200000 + 0.05 * 200000 + 0.075 * 100000 + 0.1 * 100000;
    }
    printf("%.2lf\n", ret);
    return 0;
}
