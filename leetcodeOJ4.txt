void split_url(string& url)
{
	size_t i1 = url.find(':');
	if (i1 != string::npos)
		cout << url.substr(0, i1) << endl;
	size_t i2 = url.find('/', i1 + 3);
	if (i2 != string::npos)
		cout << url.substr(i1 + 3, i2 - i1 - 3) << endl;
	cout << url.substr(i2 + 1) << endl;
}

//题目： 仅仅反转字母：给定一个字符串 S，返回 “反转后的” 字符串，其中不是字母的字符都保留在原地，而所有字母的位置发生反转。
class Solution {
public:
    bool IsChar(char ch)
    {
        if( (ch <='Z' && ch >='A') || (ch <='z' && ch >='a'))
        return true;
        else
        return false;
    }
    string reverseOnlyLetters(string s) {
        int left=0;
        int right=s.size()-1;
        while(left < right)
        {
            while(left < right && IsChar(s[left])==false)  
            {
                left++;
            }
            while(left < right && IsChar(s[right])==false)  
            {
                right--;
            }
            swap(s[left],s[right]);
            ++left;
            --right;
        }
        return s;
    }
};

//题目： 字符串中的第一个唯一字符：给定一个字符串，找到它的第一个不重复的字符，并返回它的索引。如果不存在，则返回 -1。
class Solution {
public:
    int firstUniqChar(string s) {
        int count[26]= {0};
          for(auto ch:s)       //映射的方式统计每个字符出现的次数
          {
              count[ch-'a']++;
          }
          for(size_t i=0;i<s.size();++i)   //遍历统计数组count，把首次出现等于1的数组线标返回
          {
              if(count[s[i]-'a']==1)  
              return i;
          }
          return -1;
    }
};

//题目： 字符串最后一个单词的长度：
描述
计算字符串最后一个单词的长度，单词以空格隔开，字符串长度小于5000。

输入描述：
输入一行，代表要计算的字符串，非空，长度小于5000。

输出描述：
输出一个整数，表示输入字符串最后一个单词的长度。

#include<iostream>
#include<stdio.h>
using namespace std;

int TheLastWordNumsSize(string s)
{
	int cur = 0;
	int count = 0;
	while (cur < s.size())
	{
		count = 0;
		while (cur < s.size() && s[cur] != ' ')
		{
			cur++;
			count++;
		}
		if (cur >= s.size())
			return count;
		while (++cur < s.size() && s[cur] == ' ')  //一致循环到结束或者下个不是空格字符
			;
	}
	return count;
}
int main()
{
	string s;
	char ch;
	while ((ch = getchar()) != '\n')
	{
		s.push_back(ch);
	}

	cout << TheLastWordNumsSize(s) << endl;
	return 0;
}

int GetLastWordSize(string& line)
{
    int count=0;
    string::reverse_iterator rit=line.rbegin();
    while(*rit==' ')
        rit++;
    while(*rit!=' ' && rit!=line.rend())
    {
        ++rit;
        ++count;
    }
    return count;
}

//题目：验证回文串：给定一个字符串，验证它是否是回文串，只考虑字母和数字字符，可以忽略字母的大小写。

说明：本题中，我们将空字符串定义为有效的回文串。
class Solution {
public:
    bool IsNumOrLetter(char& ch)
    {
        if(ch>='0' && ch<='9')
        return true;
        else if(ch<='z' && ch >= 'a')
        {
            ch-=32;
            return true;
        }
        else if(ch<='Z' && ch>='A')
        return true;
        else
        return false;
    }
    bool isPalindrome(string s) {
        int left = 0 ,right = s.size()-1;
        while(left<right)
        {
            while(left<right && IsNumOrLetter(s[left])==false)
            ++left;
            while(left<right && IsNumOrLetter(s[right])==false)
            --right;

            if(s[left] != s[right])
            return false;
            else
            {
                ++left;
                --right;
            }
        }
        return true;
    }
};

 value1 = num1[end1--]-'0';

 value1 = 0;
 
 if(end2 >= 0)
 value2 = num2[end2--]-'0';
 else
 value2 = 0;
 
 int valueret = value1 + value2 + next;
